```
try:
    # 提示用户输入一个整数
    num = int(input("请输入一个整数："))

    # 使用8除以用户输入的整数并输出
    result = 8 / num

    print(result)

# 预判到可能的错误时
except ValueError:
    print("请输入正确的整数")

# 无法预判到还会出现什么错误
except Exception as result:
    print("未知错误：%s" % result)
    
else:
    print("输出成功时才会执行的代码")

finally:
    print("无论成功与否，都会执行的代码")
```
